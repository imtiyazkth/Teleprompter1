How to Set Up

Step-by-step: integrate the converter, set up PWA offline mode, and install on your phone.

Good news: This uses the single-threaded ffmpeg.wasm core — no SharedArrayBuffer required. Works on all modern mobile browsers including Android Chrome and Brave.
1
Click "Load FFmpeg"
This downloads the FFmpeg WebAssembly core (~30MB) from the CDN. It's cached by the browser after the first load, so subsequent visits are instant. The WASM runs 100% locally — your files never leave your device.
2
Drop or Select your WebM file
Tap the drop zone and choose your recorded .webm file. This is the format produced by the browser's MediaRecorder API (used in PrompterPro). Files stay in your browser's memory only.
3
Click "Convert to MP4"
FFmpeg runs the conversion command: ffmpeg -i input.webm -c:v copy -c:a aac output.mp4. This remuxes the video stream and re-encodes audio to AAC — fast and lossless for video quality.
4
Download your MP4
The converted MP4 is created in memory as a Blob and offered for download. On Android, it saves to your Downloads folder. Compatible with WhatsApp, Instagram, YouTube, and all video platforms.

Conversion JavaScript — Add to <script>
JavaScript
/* ════════════════════════════════════════
   WebM → MP4 CONVERTER (ffmpeg.wasm)
   Add this block inside your script tag
════════════════════════════════════════ */

let ffmpegInstance = null;
let ffmpegLoaded = false;

// Load ffmpeg (call once, reuse after)
async function loadFFmpeg() {
  if (ffmpegLoaded) return ffmpegInstance;

  const { FFmpeg } = FFmpegWASM;
  const { toBlobURL } = FFmpegUtil;

  ffmpegInstance = new FFmpeg();

  // Show progress during core load
  ffmpegInstance.on('log', ({ message }) => {
    console.log('[ffmpeg]', message);
  });

  // Load single-threaded core (no SharedArrayBuffer needed)
  await ffmpegInstance.load({
    coreURL: await toBlobURL(
      'https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.js',
      'text/javascript'
    ),
    wasmURL: await toBlobURL(
      'https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.wasm',
      'application/wasm'
    ),
  });

  ffmpegLoaded = true;
  return ffmpegInstance;
}

// Convert recorded WebM blob to MP4
async function convertToMP4(webmBlob) {
  const convertBtn = document.getElementById('btn-convert-mp4');
  const progressDiv = document.getElementById('mp4-progress');
  const pctSpan     = document.getElementById('mp4-pct');
  const fillDiv     = document.getElementById('mp4-fill');

  convertBtn.textContent = 'Loading FFmpeg...';
  convertBtn.disabled = true;

  try {
    const ffmpeg = await loadFFmpeg();
    const { fetchFile } = FFmpegUtil;

    convertBtn.textContent = 'Converting...';
    progressDiv.style.display = 'block';

    // Track conversion progress
    ffmpeg.on('progress', ({ progress }) => {
      const pct = Math.round(progress * 100);
      pctSpan.textContent = pct + '%';
      fillDiv.style.width = pct + '%';
    });

    // Write input file to ffmpeg's virtual filesystem
    await ffmpeg.writeFile('input.webm', await fetchFile(webmBlob));

    // Run conversion: copy video, encode audio to AAC
    await ffmpeg.exec([
      '-i', 'input.webm',
      '-c:v', 'copy',      // copy video stream (no re-encode)
      '-c:a', 'aac',       // convert audio to AAC
      '-b:a', '128k',      // audio bitrate
      '-movflags', '+faststart', // optimize for streaming
      'output.mp4'
    ]);

    // Read output file from virtual FS
    const data = await ffmpeg.readFile('output.mp4');
    const mp4Blob = new Blob([data.buffer], { type: 'video/mp4' });
    const url = URL.createObjectURL(mp4Blob);

    // Trigger download
    const a = document.createElement('a');
    a.href = url;
    a.download = 'recording-' + Date.now() + '.mp4';
    a.click();

    // Cleanup virtual FS
    await ffmpeg.deleteFile('input.webm');
    await ffmpeg.deleteFile('output.mp4');

    convertBtn.textContent = '✓ MP4 Downloaded!';
    toast('MP4 conversion complete ✓');

  } catch (err) {
    console.error(err);
    toast('⚠ Conversion failed: ' + err.message);
    convertBtn.textContent = '🎬 Convert to MP4';
    convertBtn.disabled = false;
  } finally {
    progressDiv.style.display = 'none';
    pctSpan.textContent = '0%';
    fillDiv.style.width = '0%';
  }
}

// Wire up the button (after video modal is shown)
document.getElementById('btn-convert-mp4')
  .addEventListener('click', () => {
    if (state.recBlob) convertToMP4(state.recBlob);
    else toast('⚠ No recording found');
  });

A Progressive Web App (PWA) makes your teleprompter installable on any device and usable offline. You need 3 files in the same folder.

📁 your-project/
  ├── index.html  ← your main app
  ├── manifest.json  ← NEW: PWA identity
  ├── sw.js  ← NEW: offline service worker
  └── icon-192.png  ← app icon (192×192px)
1
Create manifest.json
This file defines your app's name, icon, and how it looks when installed. Copy the code from the "manifest.json" tab and save as manifest.json in your project root.
2
Create sw.js
The service worker handles offline caching. Copy from the "sw.js" tab and save as sw.js in your project root (same folder as index.html).
3
Add 3 lines to your HTML <head>
Link the manifest and register the service worker. Add these lines inside your <head> tag:
index.html — <head> additions
<!-- PWA Manifest -->
<link rel="manifest" href="manifest.json">
<meta name="theme-color" content="#07090d">
<meta name="mobile-web-app-capable" content="yes">
<meta name="apple-mobile-web-app-capable" content="yes">
<meta name="apple-mobile-web-app-status-bar-style" content="black-translucent">
4
Register Service Worker in JavaScript
Add this script just before your closing </body> tag or inside your existing script:
Register Service Worker
// Register Service Worker for PWA offline support
if ('serviceWorker' in navigator) {
  window.addEventListener('load', async () => {
    try {
      const reg = await navigator.serviceWorker.register('./sw.js');
      console.log('SW registered:', reg.scope);
    } catch (err) {
      console.warn('SW registration failed:', err);
    }
  });
}
5
Deploy to GitHub Pages
Push all files to your GitHub repo. Go to Settings → Pages → Source: main branch. Your app will be live at username.github.io/repo-name and installable as a PWA.
Important: PWA service workers only work on HTTPS. GitHub Pages provides free HTTPS automatically. Local testing works via localhost.

Save this as sw.js in your project root. This service worker caches your app for offline use.

sw.js — Service Worker
JavaScript
/* ══════════════════════════════════════════════════
   PrompterPro — Service Worker
   Handles: Offline caching, PWA install, updates
   Created by Imtiyaz Surjapuri
══════════════════════════════════════════════════ */

const CACHE_NAME = 'prompterpro-v1';
const CACHE_VERSION = 1;

// Files to cache for offline use
const STATIC_ASSETS = [
  '/',
  './index.html',
  './manifest.json',
  './icon-192.png',
  './icon-512.png',
  // Add your converter page if separate:
  // './converter.html',
];

// External resources to cache (fonts, etc.)
const EXTERNAL_CACHE = [
  'https://fonts.googleapis.com/css2?family=Bebas+Neue&family=Source+Serif+4:opsz,wght@8..60,300;8..60,400&family=JetBrains+Mono:wght@400;500&display=swap',
];

/* ── INSTALL: Cache all static assets ── */
self.addEventListener('install', event => {
  console.log('[SW] Installing...');
  event.waitUntil(
    caches.open(CACHE_NAME).then(async cache => {
      // Cache static assets (fail silently on errors)
      await cache.addAll(STATIC_ASSETS).catch(err =>
        console.warn('[SW] Static cache partial fail:', err)
      );
      // Cache external fonts (optional, fail silently)
      for (const url of EXTERNAL_CACHE) {
        await cache.add(url).catch(() => {});
      }
      console.log('[SW] Assets cached ✓');
    })
  );
  // Activate immediately without waiting
  self.skipWaiting();
});

/* ── ACTIVATE: Clean up old caches ── */
self.addEventListener('activate', event => {
  console.log('[SW] Activating...');
  event.waitUntil(
    caches.keys().then(keys =>
      Promise.all(
        keys
          .filter(key => key !== CACHE_NAME)
          .map(key => {
            console.log('[SW] Deleting old cache:', key);
            return caches.delete(key);
          })
      )
    ).then(() => self.clients.claim())
  );
});

/* ── FETCH: Serve from cache, fall back to network ── */
self.addEventListener('fetch', event => {
  const { request } = event;
  const url = new URL(request.url);

  // Skip non-GET requests
  if (request.method !== 'GET') return;

  // Skip chrome-extension, data: URLs
  if (!url.protocol.startsWith('http')) return;

  // Strategy: Cache First for same-origin, Network First for CDN
  if (url.origin === location.origin) {
    // CACHE FIRST for local files
    event.respondWith(
      caches.match(request).then(cached => {
        if (cached) return cached;
        // Not in cache — fetch from network and cache it
        return fetch(request).then(response => {
          if (!response || response.status !== 200) return response;
          const responseClone = response.clone();
          caches.open(CACHE_NAME).then(cache =>
            cache.put(request, responseClone)
          );
          return response;
        }).catch(() => {
          // Offline fallback for HTML pages
          if (request.headers.get('accept')?.includes('text/html')) {
            return caches.match('./index.html');
          }
        });
      })
    );
  } else {
    // NETWORK FIRST for CDN resources (fonts, ffmpeg.wasm)
    event.respondWith(
      fetch(request)
        .then(response => {
          if (!response || response.status !== 200) return response;
          const responseClone = response.clone();
          caches.open(CACHE_NAME).then(cache =>
            cache.put(request, responseClone)
          );
          return response;
        })
        .catch(() => caches.match(request)) // serve cached if offline
    );
  }
});

/* ── MESSAGE: Handle skip-waiting from UI ── */
self.addEventListener('message', event => {
  if (event.data?.action === 'skipWaiting') {
    self.skipWaiting();
  }
});
Update strategy: When you push new code to GitHub, users will get the new version on their next visit. The service worker detects the change, caches the new files, and applies them on the next page load automatically.
Once your app is on GitHub Pages with manifest.json and sw.js in place, here's how to install it on Android.
1
Open in Chrome or Brave on Android
Navigate to your GitHub Pages URL: https://yourusername.github.io/repo-name. Chrome and Brave are the best supported browsers for PWA install on Android.
2
Tap the "Install App" Banner
After a few seconds, Chrome shows a banner at the bottom: "Add PrompterPro to Home Screen". Tap Install. If the banner doesn't appear, use the manual method below.
3
Manual: Tap ⋮ Menu → Add to Home Screen
In Chrome: tap the three-dot menu (⋮) at the top right → scroll down to "Add to Home Screen" → tap Add. In Brave: same menu → "Install app".
4
App icon appears on Home Screen
PrompterPro now appears on your home screen like a native app. It opens in fullscreen mode with no browser address bar. It works offline after the first load.
5
Test Offline Mode
Turn on Airplane Mode on your phone. Open the app from the home screen. It should load completely from cache — all scripts, the teleprompter, and stored scripts work offline. Camera and recording work offline too. Only the ffmpeg.wasm CDN needs internet on first use.
Tip — Pre-cache ffmpeg.wasm for offline: After installation, open the converter once with internet, click "Load FFmpeg", and wait for it to finish. The service worker caches the WASM files. After that, the converter works offline too.
